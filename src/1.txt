fn gcd(a: int, b: int) -> int {
    if a > b {
        let temp: int = a;
        a = b;
        b = temp;
    }
    while a != 0 {
        while a <= b {
            b = b - a;
        }
        let temp1: int = b;
        b = a;
        a = temp1;
    }
    return b;
}
